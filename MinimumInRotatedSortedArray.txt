class Solution:
    def findMin(self, nums: List[int]) -> int:
        res=nums[0] #set to any default value initially
        #binary search
        l, r = 0, len(nums)-1

        while l<=r:
            if nums[l] < nums[r]:
                res=min(res,nums[l])
                break
            
            m=(l+r)//2
            res=min(res, nums[m])
            
            if nums[m] >= nums[r]:  #if mid value greater than all values  on right then right portion of sorted array is selected for further
                l= m+1
            else:
                r=m-1
        return res
