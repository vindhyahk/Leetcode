class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        
        stones=[-s for s in stones]
        heapq.heapify(stones)		#implement the maxheap from python minheap convert by taking -ves

        while len(stones) > 1:
            first = heapq.heappop(stones)
            second = heapq.heappop(stones)

            if second > first:
                heapq.heappush(stones, first-second)

        stones.append(0)
        return abs(stones[0])
