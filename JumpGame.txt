class Solution:
    def canJump(self, nums: List[int]) -> bool:
        goal=len(nums)-1
        
        for i in range(len(nums)-1, -1, -1):
            if i + nums[i] >= goal:
                goal=i
            
        if goal==0:
            return True
        else:
            return False

# T.C : O(n)
# we check the array from the last element upto the first and use greedy approach to check whether the jump can be satisfied or not
