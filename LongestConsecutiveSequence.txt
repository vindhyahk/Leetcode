class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        numSet=set(nums)
        longest=0

        for n in numSet:
            if (n-1) not in numSet:
                length=1            #100->
                while (n+length) in numSet:
                    length+=1       #1->2->..
                longest=max(length, longest)
        return longest

        # T.C: O(n) S.C: O(n)
