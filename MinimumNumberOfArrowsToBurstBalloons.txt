class Solution:
    def findMinArrowShots(self, points: List[List[int]]) -> int:
        points.sort()      # Step 1: Sort balloons by their start coordinate
        
        res = len(points)  # Step 2: Start by assuming each balloon needs its own arrow
        prev = points[0]   # Step 3: Keep track of the previous interval

        for i in range(1, len(points)):
            curr = points[i]

            # Step 4: Check if current balloon overlaps with the previous one
            if curr[0] <= prev[1]:
                res -= 1   # Overlap found → merge → one less arrow needed
                prev = [curr[0], min(curr[1], prev[1])]  # Merge overlapping range
            else:
                prev = curr  # No overlap → move to next group

        return res
