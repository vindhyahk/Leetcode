# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
        
        cur=root #start at root exists always

        while cur:
            if p.val > cur.val and q.val > cur.val :    #if values greater then search right side of BST
                cur= cur.right
            elif p.val < cur.val and q.val < cur.val :
                cur=cur.left
            else:
                return cur
