class Solution:
    def wordPattern(self, pattern: str, s: str) -> bool:
        words=s.split()

        if len(pattern)!=len(words):	#check if length of words and pattern equal
            return False

        chartow,wtochar={},{}		#create 2 dictionaries

        for c,w in zip(pattern, words):
            
            if c in chartow and chartow[c]!=w: 
                return False
            if w in wtochar and wtochar[w]!=c:
                return False

            chartow[c]=w
            wtochar[w]=c
        return True
