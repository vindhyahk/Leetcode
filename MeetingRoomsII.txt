"""
Definition of Interval:
class Interval(object):
    def __init__(self, start, end):
        self.start = start
        self.end = end
"""

class Solution:
    def minMeetingRooms(self, intervals: List[Interval]) -> int:
	
	start = sorted([i.start for i in intervals])
	end = sorted([i.end for i in intervals])
	
	res,count=0,0
	s,e=0,0	#pointers for start and end sorted arrays
	
	while s < len(intervals):	#wkt start array will always end first so
		if start[s] < end[e]:
			s+=1
			count+=1
		else:
			e+=1
			count-=1
		res=max(res, count)
	return res
