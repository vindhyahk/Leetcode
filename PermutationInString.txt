class Solution:
    def checkInclusion(self, s1: str, s2: str) -> bool:
        
        # If s1 is longer than s2, permutation cannot exist
        if len(s1) > len(s2):
            return False

        # Dictionaries to store character frequencies
        s1Count, s2Count = {}, {}

        # Build frequency map for s1
        for c in s1:
            s1Count[c] = 1 + s1Count.get(c, 0)

        # Left pointer of sliding window
        l = 0

        # Iterate through s2 with right pointer
        for r in range(len(s2)):

            # Add current right character to window count
            s2Count[s2[r]] = 1 + s2Count.get(s2[r], 0)

            # If window size becomes larger than s1, shrink from the left
            if r - l + 1 > len(s1):

                # Remove the left character from window count
                s2Count[s2[l]] -= 1

                # If frequency becomes 0, remove key from dictionary
                if s2Count[s2[l]] == 0:
                    del s2Count[s2[l]]

                # Move left pointer forward
                l += 1

            # If both frequency maps match, permutation exists
            if s1Count == s2Count:
                return True

        # No permutation of s1 found in s2
        return False
