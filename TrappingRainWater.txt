class Solution:
    def trap(self, height: List[int]) -> int:

        sum_water=0
        max_right=[0]
        max_left=[0]

        max_right_height=0
        max_left_height=0

        for i in range(1, len(height)-1):
            if height[i-1] > max_left_height:
                max_left_height = height[i-1]
                max_left.append(max_left_height)
            else:
                max_left.append(max_left_height)
        
        for i in range(len(height)-2, -1, -1):
            if height[i+1] > max_right_height:
                max_right_height = height[i+1]
                max_right.append(max_right_height)
            else:
                max_right.append(max_right_height)

        max_right = max_right[::-1]
        
        for i in range(1, len(height)-1):
            stored_water=min(max_left[i],max_right[i]) - height[i]

            if stored_water>0:
                sum_water+=stored_water
        return sum_water
