MAX_CHAR=26
class Solution:
    def nonRepeatingChar(self,s):
        #using frequency array

        freq=[0]*MAX_CHAR
        
        for c in s:
            freq[ord(c) - ord("a")] +=1
            
        for c in s:
            if freq[ord(c) - ord("a")] == 1:
                return c
            
        return '$'
    
    # T.C : O(n) S.C : O(n)
