class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        res=[]

        for i in range(len(intervals)):         #iterate through given interval list
            if newInterval[1] < intervals[i][0]:    
#if rhs value of newinterval lesser than first value of the interval then nonoverlapping just append and add the interval to res

                res.append(newInterval)
                return res + intervals[i: ]
            
            elif newInterval[0] > intervals[i][1]:
                res.append(intervals[i])
            
            else:
                newInterval = [ min(newInterval[0], intervals[i][0]), max(newInterval[1], intervals[i][1])]

#min of lhs interval and max of rhs interval
        res.append(newInterval)

        return res

        #T.C : O(n)
        #S.C : O(n)
