class Solution:
    def isValid(self, s: str) -> bool:
        stack=[]
        closeToOpen = {")" :"(", "]" : "[", "}" : "{"}

        for c in s:
            if c in closeToOpen:    #meaning it is a closing bracket then
                if stack and stack[-1] == closeToOpen[c]:   #check if stack not empty and top of stack matches the bracket then pop
                    stack.pop()
                else:
                    return False
            else:
                stack.append(c)
        return True if not stack else False
