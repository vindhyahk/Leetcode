class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:
        if sum(gas)<sum(cost):
            return -1
        
        total,res=0,0

        for i in range(0,len(gas)):             #gas    1   2  3 4 5
            total += gas[i] - cost[i]           #cost   3   4  5 1 2
            if total<0 :                        #diff   -2 -2 -2 3 3
                total=0
                res= i+1 
        
        return res


        # T.C : O(n)
